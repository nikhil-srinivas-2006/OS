Binary Search:
#include <stdio.h>
#include <stdlib.h>

void main() {
    int i, n, mid, l;
    printf("Enter length of array:");
    scanf("%d", &l);
    int arr[l];
    printf("Enter the elements of the array:\n");
    for(i = 0; i < l; i++) {
        scanf("%d", &arr[i]);
    }
    printf("Enter num to search:");
    scanf("%d", &n);
    int left=0, right=l-1;
    while(left <= right) {
        mid = (left + right) / 2;
        if(arr[mid] == n) {
            printf("Element found");
            break;
        }
        else if(arr[mid] < n) {
            left = mid + 1;
        }
        else {
            right = mid - 1;
        }
    }
}

Enter the elements of the array:
9 4 2 7 5
Enter num to search:7
Element found




Smallest Number:
#include <stdio.h>
#include <stdlib.h>

void main() {
    int i, l;
    printf("Enter length of array:");
    scanf("%d", &l);
    int arr[l];
    printf("Enter the elements of the array:\n");
    for(i = 0; i < l; i++) {
        scanf("%d", &arr[i]);
    }
    int s=arr[0], j;
    for(i = 0; i < l; i++) {
        if(arr[i] < s) {
            s = arr[i];
            j = i;
        }
    }
    printf("Smallest element %d at pos %d", s, j);
}

Duplicates:
#include <stdio.h>
#include <stdlib.h>

void main() {
    int i, l, j;
    printf("Enter length of array:");
    scanf("%d", &l);
    int arr[l];
    printf("Enter the elements of the array:\n");
    for(i = 0; i < l; i++) {
        scanf("%d", &arr[i]);
    }
    for(i = 0; i < l; i++) {
        for(j = i+1; j < l; j++) {
            if(arr[i] == arr[j]) {
                printf("Duplicate found");
                break;
            }
        }
    }
}
